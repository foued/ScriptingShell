#! /bin/bash/

To="rezgui.foued@gmail.com"
ram_free=$(free -mt | grep Total:|awk '{print $4}')

if [$ ram_free -le 700]
then echo "Sending mail because your ram free size is less than 700"
echo "Subject:warning,RAM free Size is LOW"| sendmail $To
fi

exit #E_XCD
